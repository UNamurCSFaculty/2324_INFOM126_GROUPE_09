createRequestOption
Default value : (req?: any): HttpParams => { let options: HttpParams = new HttpParams(); if (req) { Object.keys(req).forEach(key => { if (key !== 'sort' && req[key] !== undefined) { for (const value of [].concat(req[key]).filter(v => v !== '')) { options = options.append(key, value); } } }); if (req.sort) { req.sort.forEach((val: string) => { options = options.append('sort', val); }); } } return options; }
